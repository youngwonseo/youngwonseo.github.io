def generator(data, lookback, delay, min_index, max_index, shuffle=False, batch_size=128, step=6):
  if max_index is None:
    max_index = len(data) - delay - 1
  i = min_index + lookback #아래에서 lookback을 뺀 값 부터 현재 값 까지 사용하기 때문에 
  while 1:
    if shuffle:
      #최소, 최대, 개수
      rows = np.random.randint(min_index + lookback, max_index, size=batch_size)
    else:
      if i + batch_size >= max_index: #선택범위가 max_index를 넘어서면
        i = min_index + lookback #초기치로?
      rows = np.arange(i, min(i+batch_size, max_index)) #선택범위 i ~ i+batch_size
      i += len(rows) #batch size만큼 증가(다음 반복에서 사용)

    samples = np.zeros((len(rows),
                        lookback // step, #샘플 하나당 포함된 데이터 포인트 수
                        data.shape[-1]))
    targets = np.zeros((len(rows),))
    for j, row in enumerate(rows):
      indices = range(rows[j] - lookback, rows[j], step) #제너레이터
      samples[j] = data[indices]
      targets[j] = data[rows[j] + delay][1] #delay후 2번째 특성
    yield samples, targets

  • shuffle : 샘플을 섞을지 시간순서대로 추출할지 여부
