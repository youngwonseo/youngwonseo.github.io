자바 유니코드 한글 전환

유니코드 문자열을 한글로 변경하는 함수입니다.

public static String unicodeConvert(String str) {
	StringBuilder sb = new StringBuilder();
    char ch;
    int len = str.length();
    for (int i = 0; i < len; i++) {
	    ch = str.charAt(i);
        if (ch == '\\' && str.charAt(i+1) == 'u') {
	        sb.append((char) Integer.parseInt(str.substring(i+2, i+6), 16));
            i+=5;
	        continue;
        }
	    sb.append(ch);
	}
	return sb.toString();
}
