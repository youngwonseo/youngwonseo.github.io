자바스크립트 - 기존 function과 화살표 함수에서의 This

2020-03-17 / Youngwon Seo

ES6에서 존재하는 화살표함수와 기존의 function은 바인딩되는 this의 차이가 존재합니다.

function BlackDog() {
    this.name = '흰둥이';
    return {
        name: '검둥이',
        bark: function() {
            console.log(this.name + ': 멍멍!');
        }
    }
}


function WhiteDog() {
    this.name = '흰둥이';
    return {
        name: '검둥이',
        bark: () => {
            console.log(this.name + ': 멍멍!');
        }
    }
}

new BlackDog().bark();  //검둥이: 멍멍!
new WhiteDog().bark();  //흰둥이: 멍멍!

function은 자신이 종속된 객체, 화살표 함수는 자신이 종속된 인스턴스를 가르킴

중괄호가 없으면 바로 리턴을 의미

const result = (data) => data * 2;
