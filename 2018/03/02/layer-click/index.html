1. background

2. click event

  • we already implement layer on google map.
  • click is implmented in views/index.html

  • this is code for click event and it is focused on Busan!

    //fillColor, fillOpacity, strokeWeight..
mapData.setStyle(function(feature){
    //console.log(feature);
    //EMD_CD is localcode
    if(feature.getProperty('EMD_CD').startsWith('26')){
        //it is Busan. busan is filled blue.
        return {fillColor:'#0000ff'};
    }else{
        //it is not Busan.
        return null;
    }
});

mapData.addListener('click',function(e){
    //localcode
    if(e.feature.getProperty('EMD_CD').startsWith('26')){
        alert('this is Busan! ' + e.feature.getProperty('EMD_CD'))	
    }else{
        //nothing
    }

})
  • after occur click event, it saperate by localcode that is EMD_CD.
